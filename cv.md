```
const isPerfectSquare = function (num) {
  if (num === 1) return true;

  let start = 0;
  let end = num;
  let mid, square;

  while (start < end) {
    mid = start + parseInt((end - start) / 2);
    square = mid * mid;

    if (square === num) {
      return true;
    }

    if (square < num) {
      start = mid + 1;
    } else {
      end = mid;
    }
  }

  return false;
};
```
